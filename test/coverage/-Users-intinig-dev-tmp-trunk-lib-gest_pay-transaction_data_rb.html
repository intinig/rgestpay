 62     def initialize(attributes = nil)
 63       @separator = "*P1*"
 64       @attributes = {:encrypted_str => nil}
 65       REQUEST_MAPPINGS.merge(RESPONSE_MAPPINGS).merge(ERROR_MAPPINGS).each_value do |v|
 66         @attributes = @attributes.merge({v => nil})
 67       end
 68       if attributes.respond_to? :has_key?
 69         create_from_hash attributes
 70       elsif attributes.respond_to? :to_str
 71         create_from_string attributes.to_str
 72       end
 73     end

127     def create_from_string attributes
128       @attributes[:custom_info] = ""
129       Hash[*attributes.split(/#{Regexp.escape(@separator)}|=/)].each do |key, value|
130         if @attributes.has_key? map_attribute_key(key)
131           @attributes[map_attribute_key(key)] = CGI.unescape(value)
132         else 
133           @attributes[:custom_info] << @separator + CGI.unescape(key) + "=" + CGI.unescape(value)
134         end
135       end
136       @attributes[:custom_info] = nil if @attributes[:custom_info].size == 0
137       if @attributes[:error_code] == "0"
138         @attributes[:error_code] = nil
139         @attributes[:error_description] = nil
140       end
